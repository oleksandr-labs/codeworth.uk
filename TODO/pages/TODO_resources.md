```ts
type Resource = {
  slug: string;
  titleUk: string;
  titleEn: string;
  description: string;
  category: "checklist" | "guide" | "template" | "tool" | "report";
  format: "pdf" | "page" | "interactive";

  downloadUrl?: string;     // якщо публічний файл
```
